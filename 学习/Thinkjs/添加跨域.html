1.安装 koa-cors 库

nnpm install --save koa-cors
yarn add koa-cors


2.在文件 src/config/middlewa.jsre 中配置

// 在数组第二个
{
  handle: cors
}

3.跳过 option 请求

在 src/middleware 创建中间件 indextest.js

module.exports = (options, app) => {
  return async (ctx, next) => {
    if (ctx.method === 'OPTIONS') {
      return ctx.json({msg:'Ok'})
    }
    return next()
  }
}
