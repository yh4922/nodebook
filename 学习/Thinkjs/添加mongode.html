1.安装 think-mongoose 库

nnpm install --save think-mongoose
yarn add think-mongoose

2.在文件夹配置连接

// 使用用户授权连接
exports.model = {
  type: 'mongoose',
  mongoose: {
    handle: mongoose,
    host: '192.168.1.148',
    user: 'bbstest',
    password: '123456',
    database: 'bbstest',
    useCollectionPlural: false,
    options: {}
  }
}

// 不需要用户密码连接
