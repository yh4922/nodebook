// 随机生成姓氏
function familyName() {
    // 姓氏列表，随机从列表里面获取一个
    var familyNames = "赵钱孙李周吴郑王冯陈褚卫蒋沈韩杨朱秦尤许何吕施张孔曹严华";
    return familyNames[Math.floor((Math.random() * familyNames.length))]
}



// 随机生成手机号码
function getMoble() {
    var prefixArray = ["130", "131", "132", "133", "135", "137", "138", "170", "187", "189"] // 手机号段
    var prefix = prefixArray[Math.floor((Math.random() * prefixArray.length))]
    for (var j = 0; j < 8; j++) {
        prefix = prefix + Math.floor(Math.random() * 10);
    }
    return prefix
}
